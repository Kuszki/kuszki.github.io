95static error_t parse_opt(int key, char* arg, argp_state* state)
96{
97 struct arguments* args = (arguments*) state->input;
98
99 switch (key)
100 {
101 case 'p':
102 args->port = atoi(arg);
103 if (!args->port) argp_usage(state);
104 break;
105 case 'h':
106 args->host = arg;
107 if (args->host.empty()) argp_usage(state);
108 break;
109
110 case 'u':
111 if (args->mode != arguments::unknown) argp_usage(state);
112 else args->mode = arguments::upload;
113 break;
114 case 'd':
115 if (args->mode != arguments::unknown) argp_usage(state);
116 else args->mode = arguments::download;
117 break;
118
119 case ARGP_KEY_ARG:
120 switch (state->arg_num)
121 {
122 case 0:
123 args->file = arg;
124 break;
125 case 1:
126 args->local = arg;
127 break;
128 default:
129 argp_usage(state);
130 }
131 break;
132
133 case ARGP_KEY_END:
134 if (state->arg_num < 1 || args->file.empty() ||
135 args->mode == arguments::unknown) argp_usage(state);
136 else if (args->local.empty()) args->local = args->file;
137 break;
138
139 default: return ARGP_ERR_UNKNOWN;
140 }
141 return 0;
142}

155int main(int argc, char* argv[])
156{
157 // Wartości domyślne parametrów

161 .host = "localhost",
162 .port = 8080
163 };
164
165 // Przetwórz argumenty
166 argp_parse(&argp, argc, argv, 0, 0, &args);
167
168 CLIENT cli; // Utwórz klienta
169
170 // Nawiąż połączenie i wykonaj akcję
171 if (cli.connect(args.host, args.port)) switch (args.mode)
172 {
173 case arguments::download:
174 cli.download(args.file, args.local);
175 break;
176 case arguments::upload:
177 cli.upload(args.file, args.local);
178 break;
179 default: return -2;
180 }
181 else return -1;
182
183 return 0;
184}
