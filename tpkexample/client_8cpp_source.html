38 bool CLIENT::connect(const string& addr, const uint16_t port)
39 {
40  if (m_sock) this->disconnect();
41 
42  const string ports = to_string(port);
43  int sockfd(0);
44 
45  addrinfo hints, *servinfo, *p;
46 
47  // Uzupełnij strukturę podpowiedzi
48  memset(&hints, 0, sizeof(hints));
49  hints.ai_family = AF_INET;
50  hints.ai_socktype = SOCK_STREAM;
51 
52  cout << "Translating host...\t";
53 
54  // Pobierz informacje o hoście zgodnie z podpowiedziami
55  const int ret = getaddrinfo(addr.c_str(), ports.c_str(), &hints, &servinfo);
56 
57  if (ret != 0) return false;
58  else cout << "OK\n";
59 
60  cout << "Connecting socket...\t";
61 
62  // Wyszukaj pierwszy pasujący adres
63  for (p = servinfo; p != nullptr; p = p->ai_next)
64  {
65  // Na podstawie danych spróbuj utworzyć gniazdo
66  if ((sockfd = ::socket(p->ai_family,
67  p->ai_socktype,
68  p->ai_protocol)) == -1)
69  {
70  sockfd = 0; // Gdy błąd - przejdź dalej
71  continue; // do kolejnego elementu
72  }
73 
74  // Spróbuj nawiązać połaczenie
75  if (::connect(sockfd,
76  p->ai_addr,
77  p->ai_addrlen) == -1)
78  {
79  ::close(sockfd); // Gdy błąd - zamknij
80  sockfd = 0; // gniazdo po czym przejdź
81  continue; // dalej do kolejnego elemenut
82  }
83  else break; // Gdy wszystko OK pomiń resztę elementów
84  }
85 
86  // Zwolnij zasoby informacji o adresie
87  freeaddrinfo(servinfo);
88 
89  if (sockfd == 0) return false;
90  else cout << "OK\n";
91 
92  this->m_sock = sockfd;
93 
94  return true;
95 }

102 int CLIENT::download(const string& path, const string& dest)
103 {
104  cout << "Opening local file...\t";
105 
106  // Pobierz nazwę pliku i otwórz lokalny plik
107  const string name = filesystem::path(path).filename();
108  fstream file(dest, ios_base::out | ios_base::trunc | ios_base::binary);
109 
110  if (!file.is_open()) return -1;
111  else cout << "OK\n";
112 
113  // Wygeneruj nagłówek
114  const string header = "DOWNLOAD " + name + '\n';
115 
116  size_t count(0); // Licznik wszystkich danych
117  ssize_t rec(0); // Licznik danych w pakiecie
118 
119  cout << "Downloading file...\t";
120 
121  // Wyślij nagłówek do serwera
122  if (send_all(m_sock, header.c_str(), header.size()))
123  {
124  // Odbierz fragment pliku z serwera
125  while ((rec = ::recv(m_sock, m_buff, sizeof(m_buff), 0)) > 0)
126  {
127  file.write(m_buff, rec); // Zapisz go do pliku
128  count += rec; // Dodaj do licznika długość danych
129  }
130 
131  // W przypadku błędu podczas odbierania lub zamknięcia
132  // połaczenia przez serwer - rozłącz się
133  if (rec <= 0) disconnect();
134  }
135 
136  // Rozłącz się po wykonaniu zadania
137  this->disconnect();
138 
139  if (count) cout << "OK\n";
140  else cout << "FAIL\n";
141 
142  return count;
143 }

145 int CLIENT::upload(const string& path, const string& src)
146 {
147  cout << "Opening local file...\t";
148 
149  // Pobierz długość pliku i otwórz lokalny plik
150  const auto size = filesystem::file_size(src);
151  fstream file(src, ios_base::in | ios_base::binary);
152 
153  if (!file.is_open()) return -1;
154  else cout << "OK\n";
155 
156  // Pobierz nazwę pliku i wygeneruj nagłówek
157  const string name = filesystem::path(path).filename();
158  const string header = "UPLOAD " + name + '\n';
159 
160  size_t count(0); // Licznik wszystkich danych
161  bool fail = false; // Stan błędu
162 
163  cout << "Uploading file...\t";
164 
165  // Wyślij nagłówek do serwera
166  if (send_all(m_sock, header.c_str(), header.size()))
167  {
168  // Odczytuj plik i wysyłaj fragmenty do serwera
169  while (!file.eof() && !fail)
170  {
171  file.read(m_buff, sizeof(m_buff)); // Odczytaj fragment
172  const size_t chunk = file.gcount(); // Pobierz jego rozmiar
173 
174  if (chunk <= 0) break; // Gdy nic nie odczytano przerwij pętlę
175  else if (send_all(m_sock, m_buff, chunk)) count += chunk;
176  else break; // Gdy nue udało się wysłać danych przerwij pętlę
177  }
178  }
179 
180  // Rozłącz się po wykonaniu zadania
181  this->disconnect();
182 
183  if (count == size) cout << "OK\n";
184  else cout << "FAIL\n";
185 
186  return count;
187 }
