38 bool SERVER::start(const string& addr, const uint16_t port, const int queue)
39 {
40  static const int yes = 1; // Zmienna do ustawienia opcji `SO_REUSEADDR`
41  if (m_sock) this->stop(); // Zatrzymaj serwer, jeśli jest aktywny
42 
43  // Stwórz socket - IPv4, TCP
44  cout << "Creating socket...\t";
45  int sock = ::socket(PF_INET, SOCK_STREAM, 0);
46 
47  if (sock == -1) return false;
48  else cout << "OK\n";
49 
50  sockaddr_in sin;
51  int res;
52 
53  // Suspełnij strukturę adresu
54  memset(&sin, 0, sizeof(sin));
55  sin.sin_port = ::htons(port);
56  sin.sin_family = AF_INET;
57 
58  cout << "Filling struct...\t";
59 
60  // Konwertuj adres z łańcucha do liczby
61  res = inet_pton(AF_INET, addr.c_str(), &(sin.sin_addr));
62 
63  if (res <= 0) return false;
64  else cout << "OK\n";
65 
66  cout << "Setting options...\t";
67 
68  // Ustaw opcję ponownego użycia adresu
69  res = ::setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(int));
70 
71  if (res == -1) return false;
72  else cout << "OK\n";
73 
74  cout << "Binding address...\t";
75 
76  // Połącz gniazdo z adresem
77  res = ::bind(sock, (sockaddr*) &sin, sizeof(sin));
78 
79  if (res == -1) return false;
80  else cout << "OK\n";
81 
82  cout << "Start listening...\t";
83 
84  // Rozpocznij nasłuchiwanie
85  res = ::listen(sock, queue);
86 
87  if (res == -1) return false;
88  else cout << "OK\n\n";
89 
90  // Uzupełnij pola i dodaj gniazdo do listy `poll`
91  m_terminate = false;
92  m_sock = sock;

126 bool SERVER::loop(int timeout)
127 {
128  // Sprawdź stan połączeń pod kątem możliwości ich obsługi
129  if (poll(m_sockets.data(), m_sockets.size(), timeout) > 0)
130  {
131  // Pierwszy element na liście to serwer
132  const auto& serv = m_sockets.front();
133 
134  // Jeśli serwer jest gotowy do odczytu (czeka nowy klient)
135  if (serv.revents & POLLIN)
136  {
137  sockaddr_in sin; // Struktora pomocnicza na adres
138  socklen_t size = sizeof(sin); // Długość adresu
139 
140  // Akceptuj nowe połączenie do serwera
141  int sock = ::accept(serv.fd, (sockaddr*) &sin, &size);
142 
143  // Jeśli połączenie jest prawidłowe dodaj je do listy klientów
144  if (sock != -1) on_accept(sock);
145  }
146 
147  // Zacznij iteracje od pierwszego klienta
148  auto i = m_sockets.begin() + 1;
149 
150  // Obsługuj kolejne połączenia aż do końca listy
151  while (i != m_sockets.end())
152  {
153  // Jeśli w połączeniu wystąpił błąd/zostało zamknięte - zwolnij zasoby
154  if (i->revents & (POLLHUP | POLLERR)) i = on_disconnect(i);
155 
156  // Jeśli połączenie jest gotowe do odczytu i oczekuje się na nagłówek,
157  // pobierz jego fragment i przetwórz go w celu skompletowania nagłówka
158  else if (m_clients[i->fd].state == STATE::Waiting &&
159  i->revents & POLLIN) i = on_header(i);
160 
161  // Jeśli połączenie jest gotowe do odczytu i oczekuje się na dane pliku,
162  // pobierz kolejny fragment i zapisz go do pliku związanego z klientem
163  else if (m_clients[i->fd].state == STATE::Uploading &&
164  i->revents & POLLIN) i = on_upload(i);
165 
166  // Jeśli połączenie jest gotowe do zapisu i dostępne są dane w pliku,
167  // odczytaj kolejny fragment danych i wyślij go do oczekujacego klienta
168  else if (m_clients[i->fd].state == STATE::Downloading &&
169  i->revents & POLLOUT) i = on_download(i);
170 
171  else ++i; // Jeśli nie trzeba podejmować żadnej akcji przejdź do kolejnego klienta
172  }
173  }
174 
175  return !m_terminate; // Zwróć stan serwera - `false` gdy trzeba zakończyć serwer
176 }

192 SERVER::iterator SERVER::on_header(SERVER::iterator it)
193 {
194  auto& client = m_clients[it->fd]; // Obiekt bieżącego klienta
195 
196  // Gdy w buforze na nagłówek jest mniej niż 32 B wolnego miejsca
197  // rozszerz bufor nagłówka o dodatkowe 64 B na dane.
198  if (client.cap - client.size < 32)
199  client.resize(client.cap + 64);
200 
201  // Jeśli nie istnieje poprawny bufor na nagłówek - zakończ połaczenie
202  if (!client.buff) return on_disconnect(it);
203 
204  cout << "Recv header chunk from:\t" << it->fd << '\t';
205 
206  // Dopisz odebrane dane do bufora, przy czym pobierz maksymalnie
207  // tyle bajtów danych, ile jest wolnego miejsca w buforze
208  ssize_t rec = ::recv(it->fd,
209  client.buff + client.size,
210  client.cap - client.size,
211  0);
212 
213  cout << '(' << rec << " B" << ')' << '\n';
214 
215  // Jeśli wystąpił błąd lib zamknieto połaczenie - zakończ połaczenie
216  // W przeciwntm razie zaktualizuj rozmiar danych w buforze
217  if (rec <= 0) return on_disconnect(it);
218  else client.size += rec;
219 
220  const auto pos_start = client.buff; // Początek bufora
221  const auto pos_end = pos_start + client.size; // Koniec bufora
222  const auto pos_nl = find(client.buff, pos_end, '\n'); // Pozycja nowej linii
223 
224  // Jeśli znaleziono w buforze znak nowej linii
225  if (pos_nl != pos_end)
226  {
227  const auto pos_sp = find(pos_start, pos_nl, ' '); // Pozycja spacji
228 
229  // Jesli nie znaleziono spacji (brak parametru) - zakończ połączenie
230  // W przeciwnym razie zamień spację i znak nowej linii na '\0'
231  if (pos_sp == pos_nl) return on_disconnect(it);
232  else *pos_nl = *pos_sp = '\0';
233 
234  cout << "Completed header for:\t" << it->fd << '\t'
235  << '(' << pos_start << ':' << pos_sp+1 << ')' << '\n';
236 
237  // Oblicz ile danych znajduje się za nagłówkiem
238  // ilość = rozmiar_danych - pozycja_nl - pozycja_start - 1
239  const int left = client.size - (pos_nl - pos_start) - 1;
240 
241  // Jeśli komunikat to "UPLOAD"
242  if (strcmp(pos_start, "UPLOAD") == 0)
243  {
244  // Pobierz jedynie nazwę pliku - ścieżka zostanie odrzucona
245  const string name = filesystem::path(pos_sp + 1).filename();
246 
247  // Otwórz do zapisu plik o zadanej w parametrze nazwie i utnij go
248  client.file.open(name, ios_base::out | ios_base::trunc | ios_base::binary);
249 
250  // Jeśli nie udało sie otworzyć pliku - zakończ połączenie
251  // W przeciwnym razie zapisz dane za nagłówkiem (jesli są)
252  if (!client.file.is_open()) return on_disconnect(it);
253  else if (left > 0) client.file.write(pos_nl + 1, left);
254 
255  client.state = STATE::Uploading; // Zmień stan na odbiór pliku.
256  }
257 
258  // Jeśli komunikat to "DOWNLOAD"
259  else if (strcmp(pos_start, "DOWNLOAD") == 0)
260  {
261  // Otwórz do odczytu plik o zadanej w parametrze nazwie
262  client.file.open(pos_sp + 1, ios_base::in | ios_base::binary);
263 
264  // Jeśli nie udało sie otworzyć pliku - zakończ połączenie
265  if (!client.file.is_open()) return on_disconnect(it);
266 
267  client.state = STATE::Downloading; // Zmień stan na wysyłanie pliku.
268  it->events |= POLLOUT; // Sprawdzaj gotowość do zapisu danych
269  }
270 
271  // Jeśli nie rozpoznano komunikatu zamknij połączenie
272  else return on_disconnect(it);
273 
274  client.clean(); // Wyczyść bufor na nagłówek - nie będzie już potrzebny
275  }
276 
277  return ++it; // Zwróć iterator na kolejne połączenie
278 }

297 SERVER::iterator SERVER::on_download(iterator it)
298 {
299  auto& file = m_clients[it->fd].file; // Plik związany z klientem
300 
301  // Jeśli są jeszcze dane do odczytu to je obsłuż
302  // Jesli nie - zamknij połaczenie
303  if (!file.eof())
304  {
305  cout << "Sending file chunk to:\t" << it->fd << '\t';
306 
307  // Odczytaj fragment danych
308  file.read(m_buff, sizeof(m_buff));
309 
310  // Pobierz odczytaną liczbę bajtów
311  int rc = file.gcount();
312 
313  // Gdy nie odczytano danych - zakończ połaczenie
314  if (rc <= 0) return on_disconnect(it);
315 
316  // Wyślij wszystkie odczytane dane
317  int sd = ::send(it->fd, m_buff, rc, 0);
318 
319  cout << '(' << sd << '/' << rc << " B" << ')' << '\n';
320 
321  // Sprawdź, czy udało się wysłać dane
322  if (sd <= 0) return on_disconnect(it);
323 
324  // W przypadku niepełnego wysyłania cofnij
325  // wskaźnik pliku do prawidłowego miejsca
326  else if (sd < rc) file.seekg(sd - rc, ios::cur);
327  }
328  else return on_disconnect(it); // Zamknij połaczenie
329 
330  return ++it; // Zwróć iterator na kolejne połączenie
331 }
