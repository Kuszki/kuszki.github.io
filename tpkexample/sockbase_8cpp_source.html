45bool SOCKBASE::send_all(int sock, const char* data, size_t size)
46{
47 // Gdy są jeszcze dane do wysłania
48 while (size > 0)
49 {
50 // Wyślij brakujące dane
51 const int sc = ::send(sock, data, size, 0);
52
53 // W przypadku błędu przerwij działanie
54 if (sc <= 0) return false;
55 else
56 {
57 data += sc; // Przesuń wskaźnik na dane
58 size -= sc; // Zmniejsz liczbę pozostałych danych
59 }
60 }
61
62 return true;
63}
